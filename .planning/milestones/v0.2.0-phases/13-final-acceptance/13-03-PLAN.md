```rust
fn tabs_to_spaces(code: &str) -> String {
    code.lines()
        .map(|line| {
            let tab_count = line.chars().take_while(|c| *c == '\t').count();
            if tab_count > 0 {
                format!("{}{}", "    ".repeat(tab_count), &line[tab_count..])
            } else {
                line.to_string()
            }
        })
        .collect::<Vec<_>>()
        .join("\n")
}
```
